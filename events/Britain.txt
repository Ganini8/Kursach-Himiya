add_namespace = britain

#Ирландское копротивление - февральская революция
country_event = {
	id = britain.1
	hidden = yes
	
	trigger = {
		has_start_date < 1917.3.8
		tag = ENG
		RUS = { NOT = { has_country_flag = RUS_monarchy_not_fall } }
	}
	
	mean_time_to_happen = {
		days = 0
	}
	
	fire_only_once = yes
	option = {
		every_owned_state = {
			limit = { is_core_of = IRE NOT = { is_core_of = ROOT } }
			add_resistance = 18
		}
	}
}
